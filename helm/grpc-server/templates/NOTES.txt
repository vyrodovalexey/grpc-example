{{- else if contains "NodePort" .Values.service.type }}

Get the gRPC server URL by running:
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "grpc-server.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "gRPC server: $NODE_IP:$NODE_PORT"

{{- else if contains "LoadBalancer" .Values.service.type }}

Get the gRPC server URL by running:
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  Watch the status with: kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "grpc-server.fullname" . }}

  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "grpc-server.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "gRPC server: $SERVICE_IP:{{ .Values.service.grpcPort }}"

{{- else if contains "ClusterIP" .Values.service.type }}

The gRPC server is accessible within the cluster at:
  {{ include "grpc-server.fullname" . }}.{{ .Release.Namespace }}.svc.cluster.local:{{ .Values.service.grpcPort }}

To access the gRPC server from your local machine, run:
  kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "grpc-server.fullname" . }} {{ .Values.service.grpcPort }}:{{ .Values.service.grpcPort }}

Then connect to: localhost:{{ .Values.service.grpcPort }}

{{- end }}
